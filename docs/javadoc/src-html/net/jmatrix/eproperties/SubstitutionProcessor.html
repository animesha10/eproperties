001    package net.jmatrix.eproperties;
002    
003    import java.util.regex.*;
004    
005    import org.apache.commons.logging.*;
006    
007    /**
008     * Static methods used to substitute tokens for values.
009     * 
010     * Note: This class has some very tricky logic supporting 2 types of 
011     * substitutions (partial and complete), nested tokens (${key.${env}}),
012     * and default values.  
013     * 
014     * When changing this class you should: 
015     * a) know what you are doing
016     * b) know the desired token substitution syntax and function.
017     * c) test extensively with many tokenized properties files.
018     * 
019     * you have been warned.  Bemo, 21 July 2009.
020     */
021    public final class SubstitutionProcessor {
022       static Log log=LogFactory.getLog(SubstitutionProcessor.class);
023       
024       // This was the original (non-functional) regex.
025       //static final String TOKEN_REGEX = "\\$\\{.*?\\}";
026       // for nested token trial. This regex seems to work in parsing
027       // some standard files as well as nested token replacements... 
028       // i'm leaving it in place, pending further testing.  17 mar 2009.  bemo.
029       static final String TOKEN_REGEX="\\$\\{[^\\$\\{\\}]+\\}";
030    
031       static final String COMPLETE_TOKEN_REGEX="^"+TOKEN_REGEX+"$";
032       
033       static final Pattern PARTIAL_PATTERN = Pattern.compile(TOKEN_REGEX);
034       
035       static final Pattern COMPLETE_PATTERN = Pattern.compile(COMPLETE_TOKEN_REGEX);
036    
037       public static int MAX_RECURSIVE_SUBSTITUTIONS=10;
038       
039       public static boolean debug=false;
040       
041       /** 
042        * Public facing method.
043        * 
044        * There are 2 types of substitutions: partial and complete.  Complete 
045        * substitutions are only possible when the value is simply the token. 
046        * Partial substitutions are string only.  Complete substitutions can 
047        * replace a token with an object - like EProperties or List.
048        * 
049        * Partial Substitution: 
050        * key="This is a string with a ${token} value"
051        * 
052        * Complete Substitution:
053        * key2=${token}
054        * 
055        * The result of a partial substitution is ALWAYS a String.  The result
056        * of a complete substitution can be any of [ String, EProperties, List ].
057        * 
058        */
059       public static final String processSubstitution(String s, EProperties p) {
060          return (String)processSubstitution(s, p, String.class);
061       }
062       
063       public static final boolean containsTokens(String s) {
064          Matcher matcher=PARTIAL_PATTERN.matcher(s);
065          return matcher.find();
066       }
067       
068       /** */
069       public static final Object processSubstitution(String s, EProperties p, 
070                                                      Class returnType) {
071          return processSubstitution(s, p, returnType, 0);
072       }
073       
074       ///////////////////////////////////////////////////////////////////////
075       /////////////////////////  PRIVATE METHODS  ///////////////////////////
076       ///////////////////////////////////////////////////////////////////////
077       
078       
079       /** */
080       private static final boolean isCompleteToken(String s) {
081          Matcher matcher=COMPLETE_PATTERN.matcher(s);
082          return matcher.matches();
083       }
084       
085       /** */
086       private static final Object processSubstitution(String s, EProperties p, 
087                                                       Class returnType, int iter) {
088          if (returnType == null)
089             returnType=Object.class;
090          
091          if (returnType.equals(String.class)) {
092             Object rval= recursiveReplace(s, p, iter);
093             if (rval == null)
094                return rval;
095             
096             if (rval instanceof String) 
097                return rval;
098             else {
099                log.warn("processSubstitution called with String.class, "+
100                      "but result is "+rval.getClass().getName());
101                return s;
102             }
103          } else {
104             if (isCompleteToken(s)) {
105                return completeReplace(s, p, iter);
106             } else {
107                return recursiveReplace(s, p, iter);
108             }
109          }
110       }
111       
112       /**
113        * This is replacing a complete key - but should work the same as
114        * the partial replace if the value returned by the token lookup is a
115        * String.  */
116       private static final Object completeReplace(String s, EProperties p, 
117                                                   int iter) {
118          log.debug("Complete replace of token '"+s+"'");
119          
120          SubstitutionToken subToken=new SubstitutionToken(s);
121          String key=subToken.getKey();
122          String def=subToken.getDefault();
123          
124          Object val=p.findValue(key);
125          
126          if (val == null) {
127             return def;
128          } else {
129             if (val instanceof String) {
130                String sval=(String)val;
131                log.debug("Complete replace, result is string '"+sval+"'");
132                if (containsTokens(sval)) {
133                   return processSubstitution(sval, p, Object.class, iter);
134                }
135             }
136             return val;
137          }
138       }
139       
140       /**
141        * This is replacing a partial key.
142        */
143       private static final Object recursiveReplace(String s, EProperties p, int iter) {
144          // the word 'recursive' in this sense is really a recursion to support
145          // nested tokens.  Which is why it cannot fix the other, different
146          // recursion problem noted below.
147          // The recursion here is to support nested keys, like this:
148          // env=test
149          // foo=${env->${env}}
150          // Above, 'foo' has a value that is nested.  the 'recursion' of this 
151          // method is to support such nesting.  In the above case, we would
152          // go thru iter: 0, 1, 2 (where 2 doesn't really do anything).
153          
154          
155          // FIXME: This class allows recursive replace - which should not
156          // be allowed!!  
157          // Here, recursive replace is defined as the following:
158          //   key.a=${key.b}
159          //   key.b=${key.a}
160          // Which is valid syntax.  (and more common when we consider included
161          // properties files).
162          // 
163          // The real solution is not the iter parameter, but a
164          // ThreadLocal variable that counts the recursion.
165          //
166          // The key point here though is that we'd have a better error message
167          // but it would still be an error.  It will error either way.  If
168          // we don't put in the thread local counter, it will result in a 
169          // stack overflow.
170          
171          if (iter > MAX_RECURSIVE_SUBSTITUTIONS)
172             throw new RuntimeException("Error: recursive replacement limit on '"+
173                   s+"' "+iter+" exceeds max "+MAX_RECURSIVE_SUBSTITUTIONS);
174          
175          // Note: Pattern instance is immutable and thread safe.
176          Matcher matcher = PARTIAL_PATTERN.matcher(s);
177          
178          String newVal = s;
179          while (matcher.find()) {
180             log.debug("replace: found tokens in '"+s+"'");
181             
182             // value=something ${key|default} something else
183             // token will be '${key|default}
184             String token = s.substring(matcher.start(), matcher.end());
185             log.debug("   Token: '"+token+"'");
186             
187             SubstitutionToken subToken=new SubstitutionToken(token);
188             String key=subToken.getKey();
189             String def=subToken.getDefault();
190             
191             
192             String replaceVal = p.findProperty(key);
193    
194             if (replaceVal != null) {
195                log.debug("   Found value for token "+token+": "+replaceVal);
196                newVal = newVal.replace(token, replaceVal);
197             } else {
198                if (def != null) {
199                   log.debug("   No value for token "+token+", using default: "+def);
200                   newVal=newVal.replace(token, def);
201                }
202                else {
203                   // nothing - this will fail by design.
204                }
205             }
206          }
207    
208          if (newVal.equals(s))
209             return newVal; // no change
210          else
211             return processSubstitution(newVal, p, null, iter+1);
212       }
213       
214       /** */
215       static class SubstitutionToken {
216          private String key=null;
217          private String def=null;
218          
219          /** 
220           * Constructor takes a complete token, in the following form: 
221           * ${key} or ${key|def}
222           * ie, the default value is optional.
223           */
224          public SubstitutionToken(String s) {
225             String tokenKey = s.substring(2, s.length() - 1);
226             int pipeindex=tokenKey.indexOf("|");
227             if (pipeindex != -1) {
228                key=tokenKey.substring(0, pipeindex);
229                def=tokenKey.substring(pipeindex+1);
230             } else {
231                key=tokenKey;
232             }
233          }
234          public String getKey() {
235             return key;
236          }
237          public String getDefault() {
238             return def;
239          }
240       }
241    }
