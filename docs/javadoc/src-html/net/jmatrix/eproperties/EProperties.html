138       /**
139        * Flattens the property structure.  Nested properties are prefixed
140        * with the name of the nesting structure.  For instance, if we have:
141        * 
142        * foo=bar
143        * blah={
144        *   bing=bat
145        * }
146        * 
147        * Flatten would flatten the structure to:
148        * foo=bar
149        * blah|bing=bat
150        * 
151        * The delimiter is configurable (inbound on the method) with the 
152        * default delimeter being the pipe character: '|'.
153        * 
154        * */
155       protected EProperties flatten(String prefix, String delim) {
156          EProperties flat=new EProperties();
157          for (Key key:keys) {
158             Object val=get(key);
159             if (val instanceof EProperties) {
160                EProperties epval=(EProperties)val;
161                // recursion
162                EProperties nestedFlat=null;
163                if (!empty(prefix))
164                   nestedFlat=epval.flatten(prefix+delim+key.toString(), delim);
165                else 
166                   nestedFlat=epval.flatten(key.toString(), delim);
167                
168                flat.addAll(nestedFlat);
169             } else if (val instanceof List) {
170                if (!empty(prefix))
171                   flat.put(prefix+delim+key, val.toString());
172                else 
173                   flat.put(key, val.toString());
174             } else {
175                if (!empty(prefix))
176                   flat.put(prefix+delim+key, val.toString());
177                else 
178                   flat.put(key, val.toString());
179             }
180          }
181          return flat;
182       }

205       /** This method scans the properties tree, and returns a report of what
206        * properties have un-resolved substitutions.  */
207       public String validate() {
208          int size=keys.size();
209          
210          StringBuilder sb=new StringBuilder();
211          
212          for (int i=0; i<size; i++) {
213             Key key=keys.get(i);
214             
215             // substitutions should have happened at the point that
216             // this is called.
217             Object value=get(key);
218             
219             if (value instanceof String) {
220                String s=(String)value;
221                if (SubstitutionProcessor.containsTokens(s)) {
222                   // this is an error.  Tokens should have been 
223                   // replaced by values here.
224                   sb.append("Warning: Value for key '"+key+"' defined in terms of "+
225                         "un-resolvable substitution: '"+s+"'\n");
226                }
227             } else if (value instanceof List) {
228                List<String> list=(List<String>)value;
229                for (int j=0; j<list.size(); j++) {
230                   String lv=list.get(j);
231                   if (SubstitutionProcessor.containsTokens(lv)) {
232                      sb.append("Warning: List item value for key '"+key+"["+j+"]' defined "+
233                            "in terms of un-resolvable substitution: '"+lv+"'\n");
234                   }
235                }
236             } else if (value instanceof EProperties) {
237                EProperties props=(EProperties)value;
238                // maybe should indent here... 
239                sb.append(props.validate());
240             } else {
241                // not really sure what this object is, if not one of the 
242                // above types. I'm confused - but I'm not going to validate.
243             }
244          }
245          return sb.toString();
246       }

316       public String getPath() {
317          StringBuilder sb=new StringBuilder();
318          String key=thisKey;
319          EProperties nextParent=parent;
320          while (nextParent != null) {
321             sb.insert(0, "->"+key);
322             
323             key=parent.thisKey;
324             if (key == null)
325                key="UNKNOWN";
326             
327             nextParent=nextParent.parent;
328          }
329          return sb.toString();
330        }

332       /**
333        * This is the sole method for putting objects into the EProperties object.
334        * 
335        * This is synchronized to prevent concurrent access with load/read operations.
336        */
337       @Override
338       public synchronized Object put(Object k, Object v) {
339          Key key = null;
340          // 
341          if (k instanceof String) {
342             key = new Key((String) k);
343          } else if (k instanceof Key) {
344             key = (Key) k;
345          } else {
346             throw new Error("Keys in EProperties must be [Key | String].");
347          }
348          
349          String keyString=key.keyString();
350          
351          if (keyString.indexOf("->") != -1) {
352             // this is a complex key.
353             return putWithComplexKey(keyString, v);
354          }
355          
356          // Prevent duplicate keys, preserve order, notify listeners
357          int keyIndex = keys.indexOf(key);
358          if (keyIndex == -1)
359             keys.add(key);
360          else {
361             //Key existingKey = keys.get(keys.indexOf(key));
362             //existingKey.notifyListeners(v);
363             key=keys.get(keyIndex); // replace created key w/ existing key
364          }
365          
366          // build a value object.
367          // There are 3 and only 3 types of values:
368          //   1) String values.
369          //   2) List Values, and
370          //   3) EProperties values.
371          // 
372          // attempting to put any
373          
374          Object returnVal=v;
375          Value value=null;
376          if (v == null) {
377             // Allowing us to call put(key, null). Setting a property to 
378             // null is defined as removing that property.
379             //super.put(key, v);
380             super.remove(key);
381          } else if (v instanceof String) {
382             value=new StringValue((String)v, this);
383             super.put(key, value);
384          } else if (v instanceof List) {
385             value=new ListValue((List)v, this);
386             super.put(key, value);
387          } else if (v instanceof EProperties) {
388             super.put(key, v);
389             
390             // set parent.
391             //setParent((EProperties)v, key.toString());
392             ((EProperties)v).setParent(this, key.toString());
393          } else if (v instanceof StringValue ||
394                     v instanceof ListValue) {
395             super.put(key, v);
396          } else {
397             throw new Error("EProperties put() values can only be: "+
398                   "[String | List<String> | EProperties | StringValue | ListValue].  "+
399                   "Not "+v.getClass().getName());
400                            
401          }
402    
403          if (this.listeners != null) {
404             notifyListeners(key, value);
405          }
406          // This will do nothing if there are no listeners.
407          key.notifyListeners(returnVal);
408    
409          return returnVal;
410       }

412       /** 
413        * putWithCompleKey() works much like getWithComplexKey().  I will give a 
414        * simple example under 2 different conditions.
415        * 
416        * In both examples, the key is nested->foo.  In both cases the put method
417        * has been called on a properties object we'll call 'root'.  
418        * 
419        * Case 1: The 'root' EProperties object already has a nested properties
420        *         object named 'nested'.  In this case, we simply get a handle
421        *         to the deeper properties object called 'nested' and on it
422        *         we call put(key, val) with the key being a simple key 'foo' 
423        *         and the value being the inbound value.  ALL BEHAVIORS OF PUT
424        *         AT THAT POINT ARE IDENTICAL. Ie, if the value does not already 
425        *         exist, it is added.  If the value does exist, IN ANY FORM - ie
426        *         String, List or nested Properties - it is overwritten with 
427        *         the inbound object value.
428        *         
429        * Case 2: The 'root' EProperties object does not already have a nested
430        *         properties object called 'nested'.  In this case, a new properties
431        *         object is created, named 'nested' and added to 'root'.  
432        *         
433        * For keys with more than 2 levels of depth (ie, a->b->c), the process
434        * above is repeated until the last key (c in this example) is found - and 
435        * that key represtents the inbound value in the nested objects a->b that
436        * either already exist, or are created by this method.
437        */
438       private synchronized Object putWithComplexKey(String key, Object value) {
439          String keys[] = key.split("\\-\\>");
440          
441          log.debug ("getComplexKey(): "+Arrays.asList(keys));
442          
443          EProperties next = this;
444          String currentPath = "";
445          
446          //StringBuilder path=new StringBuilder();
447          
448          for (int i = 0; i < keys.length - 1; i++) {
449             // currentPath is used for debugging.
450             EProperties prev=next;
451             next=prev.getProperties(keys[i]);
452             if (next == null) {
453                // Create nested props that do not exist.
454                next=new EProperties();
455                prev.put(keys[i], next);
456             }
457          }
458          
459          return next.put(keys[keys.length-1], value);
460       }

493       @Override
494       public Object get(Object key) {
495          Object val=null;
496          
497          if (key instanceof String) { 
498             String skey=(String)key;
499             if (skey.indexOf("->") > 0)
500                val=getWithComplexKey(skey);
501             else 
502                val=super.get(key);
503          } else {
504             val=super.get(key);
505          }
506          
507          // here, we'll process substitutions if they exist...
508          if (val instanceof StringValue) {
509             Object rtval=((StringValue)val).getRuntimeValue();
510             
511             if (rtval instanceof String &&
512                SubstitutionProcessor.containsTokens((String)rtval)) {
513                log.warn("Value for key '"+key+"' defined in terms of "+
514                         "un-resolvable substitution: '"+rtval+"'");
515             }
516             return rtval;
517          }
518          else if (val instanceof ListValue) {
519             List<String> list=((ListValue)val).getRuntimeValue();
520             for (int j=0; j<list.size(); j++) {
521                String lv=list.get(j);
522                if (SubstitutionProcessor.containsTokens(lv)) {
523                   log.warn("List item value for key '"+key+"["+j+"]' defined "+
524                         "in terms of un-resolvable substitution: '"+lv+"'\n");
525                }
526             }
527             return list;
528          }
529          return val;
530       }

667       /** This is the 'master' load() method.  All other load methods delgate
668        * here for actual property loading.  */
669       public synchronized void load(InputStream is) 
670       throws IOException {
671          InputStream cis=cacheManager.getInputStream(is, sourceURL);
672          
673          EPropertiesParser parser=new EPropertiesParser(cis);
674          parser.setSourceURL(sourceURL);
675          
676          // Cache remote URLs.  Here, we need to know which URLs are local, 
677          // and which are remote.  We determine this from the sourceURL, if not
678          // null.  
679          // 
680          // The caching of remote URLs requires basically 2 'use cases': 
681          // 1) Initially, and repeatedly, we should write a cache copy of
682          //    the source properties stream.  
683          // 2) Later (at least after the first successful read), if the remote
684          //    resource is not available, we should:
685          //    a) attempt to load from the remote resource.
686          //    b) if the remote resource is not available, use the local 
687          //       cached copy.
688          // 
689          // Here, we are covering use case 1.  This is becasue all properties 
690          // loading happens at this location.
691          // unimplemented at 24 jul 09
692          try {
693             parser.parse(this);
694          } catch (Exception ex) {
695             // this constructor is only available in Java 1.6
696             //throw new IOException("Error parsing EProperties.", ex);
697             ex.printStackTrace();
698             throw new IOException("Error parsing Properties. "+ex.getMessage());
699          } finally {
700             if (cis != null) {
701                try {
702                   cis.close();
703                } catch (Exception clex) {
704                   clex.printStackTrace();
705                }
706             }
707          }
708          
709          validate();
710       }

731       /** */
732       public synchronized void load(URL url) 
733          throws IOException {
734          setSourceURL(url);
735          InputStream is=null;
736          try {
737    //         URLConnection urlConnection=url.openConnection();
738    //         
739    //         // This should work for both file:// and http:// URLs.
740    //         long tmp=urlConnection.getLastModified();
741    //         
742    //         is=url.openStream();
743             
744             is=cacheManager.getInputStream(url);
745             
746             load(is);
747    
748             if (is instanceof CacheInputStream) {
749                CacheInputStream cis=(CacheInputStream)is;
750                lastModification=cis.getLastModified();
751             }
752             //lastModification=tmp;

823       /** Save to the file currently identified by sourceURL.  sourceURL must be
824        * file based - and not http based - we can't magically save to an 
825        * http stream.
826        * 
827        * If this method is called on a properties object without a sourceURL, 
828        * this method searches up the tree to find a properties object that
829        * does have a source URL - and that object is saved if it can be found, 
830        * and has a file:// based source url.
831        * 
832        * @return Returns true if a save has been executed, false if not possible.
833        * @throws IOException
834        */
835       public boolean save() throws IOException {
836          if (sourceURL != null) {
837             if (sourceURL.getProtocol() != null
838                   && sourceURL.getProtocol().equals("file")) {
839    
840                File f = null;
841                try {
842                   f = new File(sourceURL.toURI());
843                } catch (Exception ex) {
844                   // this constructor is only available in Java 1.6
845                   //throw new IOException("Unable to form File from URL.", ex);
846                   ex.printStackTrace();
847                   throw new IOException("Unable to form File from URL. "+ex.getMessage());
848                }
849                save(f);
850             } else {
851                log.debug("Cannot save properties with URL " + sourceURL);
852             }
853             return true;
854          } else {
855             if (parent != null) {
856                return parent.save();
857             } else {
858                return false;
859             }
860          }
861       }

877       /** */
878       private synchronized void save(Writer w, int indent) 
879          throws IOException {
880          String pad="";
881          if (indent > 0) {
882             StringBuilder sb=new StringBuilder();
883             for (int i=0; i<INDENT_SPACES*indent; i++) {
884                sb.append(" ");
885             }
886             pad=sb.toString();
887          }
888          
889          try {
890             int size=keys.size();
891             for (int i=0; i<size; i++) {
892                Key key=keys.get(i);
893                
894                // writing key= is the same for string, list, properties, or
895                // inclusion.
896                
897                // first, write comment lines if they exist
898                String comments=key.getComments();
899                if (comments != null) {
900                   // trim off trailing newline, then pad, then append newline
901                   comments=comments.trim(); 
902                   
903                   comments=pad+comments;
904                   comments=comments.replace("\n", "\n"+pad);
905                   comments=comments+"\n";
906                   
907                   w.write(comments);
908                }
909                w.write(pad+key.toString()+"=");
910                
911                // now, write the value, depending on its type.
912                Object val=preSubstitutionGet(key);
913                
914                if (val == null) {
915                   w.write("\n"); // null is an unquoted empty line.
916                   // this will allow for the definition of a non-null
917                   // empty string as "".
918                } else if (val instanceof StringValue) {
919                   String s=(String) ((StringValue)val).getPersistentValue();
920                   w.write("\""+s+"\"\n");
921                } else if (val instanceof EProperties) {
922                   EProperties pval=(EProperties)val;
923                   
924                   if (pval.includeURL != null) {
925                      // these properties were included.
926                      w.write("["+pval.includeURL+"]\n");
927                   } else {
928                      w.write("{\n");
929                      pval.save(w, indent+1);
930                      w.write(pad+"}\n");
931                   }
932                } else if (val instanceof ListValue) {
933                   List list=(List)((ListValue)val).getPersistentValue();
934                   
935                   if (list.size() == 0) {
936                      w.write("( )\n");
937                   } else {
938                      
939                      String keypad=pad(key.toString().length()+2, " ");
940                      w.write("(");
941                      int lsize=list.size();
942                      for (int j=0; j<lsize; j++) {
943                         w.write("\""+list.get(j)+"\"");
944                         
945                         if (j == lsize-1) // last element
946                            w.write(")\n");
947                         else 
948                            w.write(", \n"+pad+keypad);
949                      }
950                   }
951                } else {
952                   log.error("What kind of value?? "+val.getClass().getName());
953                }
954                
955                // append this newline for whitespace readability
956                if (i != size-1)
957                   w.write("\n");
958             }
959             // end of the for loop.
960          } finally {
961             if (w != null)
962                w.flush();
963          }
964       }

1062       private Object getWithComplexKey(String key) {
1063          // a complex key uses a pointer syntax find properties
1064          // deeper in a structure.
1065          // for instance:
1066          // String s=(String)get("system->user.home");
1067          // is equivalent to
1068          // String s=(String)getProperties("system").get("user.home");
1069          //
1070          // Which brings up the issue of error handling. What if one of the
1071          // indirections returns a null nested property object!!
1072          // At that we simply return null - as if you requested a value 
1073          // from an EProperties object where the key does not exist.
1074          //
1075          // The final key can return any type of object (String, Vector,
1076          // Properties),
1077          // however all of the initial keys must return a nested EProperties
1078          // object. If they do not, then it is a runtime exception...
1079          String keys[] = key.split("\\-\\>");
1080          
1081          log.debug ("getComplexKey(): "+Arrays.asList(keys));
1082          
1083          EProperties next = this;
1084          String currentPath = "";
1085          
1086          //StringBuilder path=new StringBuilder();
1087          
1088          for (int i = 0; i < keys.length - 1; i++) {
1089             // currentPath is used for debugging.
1090             if (currentPath.equals(""))
1091                currentPath = keys[i];
1092             else
1093                currentPath = currentPath + "->" + keys[i];
1094             
1095    //         System.out.println ("getComplexKey(): i="+i+", keys[i]="+
1096    //         keys[i]+" currentPath='"+currentPath+"'");
1097            
1098             
1099             // next=next.getProperties(keys[i]);
1100             Object nextTarget = next.get(keys[i]);
1101             
1102    //         if (debug)
1103    //            System.out.println ("getComplexKey(): nextTarget is "+
1104    //               (nextTarget == null? "null":nextTarget.getClass().getName()));
1105             
1106             if (nextTarget == null) {
1107    //            if (debug)
1108    //               System.out.println("getComplexKey(): [" + key
1109    //                     + "]: Returning null because object at path "
1110    //                     + currentPath + "==null)");
1111                return null;
1112             } else if (nextTarget instanceof EProperties) {
1113                next = (EProperties) nextTarget;
1114             } else {
1115                // it is not an EProperties object!
1116    //            if (debug)
1117    //               System.out.println("getComplexKey(): [" + key
1118    //                     + "]: Returning null because object at path "
1119    //                     + currentPath + " is not EProperties, it is "
1120    //                     + nextTarget.getClass().getName());
1121                return null;
1122             }
1123          }
1124          // here, we are pointed at the correct eproperties object.
1125          return next.get(keys[keys.length-1]);
1126       }

1158       public List<String> getList(String key) {
1159          Object val = get(key);
1160          if (val == null)
1161             return null;
1162          if (val instanceof List)
1163             return (List<String>)val;
1164          
1165          // Automagically convert strings to lists.  If the return 
1166          // value is a string, it will be parsed as a csv list.
1167          // this method is quite dumb.  First, any and all substitutions
1168          // are processed when retrieving the String, not in the list.
1169          // Second, if the value is a single string - with no commas, 
1170          // a single element list with the String as the only element 
1171          // is returned.
1172          if (val instanceof String) {
1173             List<String> l=convertStringToList((String)val);
1174             return l;
1175          }
1176          
1177          return null;
1178       }

1182          List<String> l=new ArrayList<String>();
1183          String parsed[]=s.split("\\,");
1184          for (String p:parsed) 
1185             l.add(p);
1186          return l;
