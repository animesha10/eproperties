010    /** 
011     * Processes include URLs for the parser.  This contains a pluggable
012     * loader implementation for various URL formats.  
013     */
014    public class EPropertiesIncluder {
015       static Log log=LogFactory.getLog(EPropertiesIncluder.class);
016       List<PropertiesLoader> loaders=new ArrayList<PropertiesLoader>();
017       
018       PropertiesLoader urlPropsLoader=null;
019       
020       public static boolean debug=false;
021       
022       public EPropertiesIncluder() {
023          urlPropsLoader=new URLPropertiesLoader(); 
024          
025          registerLoader(urlPropsLoader);
026          //registerLoader(new SystemPropertiesLoader());
027          registerLoader(new MethodPropertiesLoader());
028       }
029       
030       /** */
031       public void registerLoader(PropertiesLoader upl) {
032          log.info("Registering loader "+upl.getClass().getName());
033          loaders.add(upl);
034       }
035       
036       /**
037        * Processes an include, typically on behalf of the parser.
038        */
039       public Object processInclude(EProperties parent, String url) {
040          // Include URL format and options are the following:
041          // [prefix://]path;[options]
042          // if [prefix://] is omitted, then it is assumed that this
043          // is a file based inclusion.  (this would be the most 
044          // common).  If the file begins with a '/', then its path
045          // is assumed to be absolute.  Otherwise, it is assumed
046          // to be relative to its parent.  
047          // file:// URLs can also be used. 
048          
049          
050          // the URL string will come in as [include.properties]
051          url=url.trim();
052          if (url.startsWith("["))
053             url=url.substring(1, url.length());
054          if (url.endsWith("]"))
055             url=url.substring(0, url.length()-1);
056          
057          // Here, we handle the potential for tokenization of the inclusion 
058          // URL.  If the token cannot be found, this will throw a 
059          // InvalidIndirectionException()
060          // this is new as of 16 feb 2009.
061          //System.out.println ("Parent properties: "+parent);
062          // parent.list(System.out);
063          if (SubstitutionProcessor.containsTokens(url))
064          url=SubstitutionProcessor.processSubstitution(url, parent);
065    
066          // if it still contains tokens, that's an error (substitution failed).
067          if (SubstitutionProcessor.containsTokens(url)) {
068             log.error("Warning: Include URL '"+url+"' defined "+
069                       "in terms of un-resolvable substitution.");
070          }
071          
072          // process options
073          EProperties options=parseOptionsFromIncludeURL(url);
074          if (options.size() > 0) {
075             // trim the options off the end of that url
076             url=url.substring(0, url.indexOf("|"));
077          }
078          
079          
080          int size=loaders.size();
081          for (int i=0; i<size; i++) {
082             PropertiesLoader upl=loaders.get(i);
083             if (upl.acceptsURL(url)) {
084                log.debug("Processing include with "+
085                     upl.getClass().getName());
086                
087                Object result=upl.loadProperties(parent, url, options);
088                //EProperties props=upl.loadProperties(parent, url, options);
089                
090                if (result instanceof EProperties) {
091                   ((EProperties)result).setIncludedURL(url);
092                }
093                return result;
094             }
095          }
096          log.debug("Hmmm... No PropertiesLoader accepts url '"+url+"'");
097          log.debug ("       Assuming URLPropertiesLoader.");
098          // if no one else claims it, assume that this URL does not 
099          // start with foo://, and assume that it is a relative URL
100          //EProperties props=urlPropsLoader.loadProperties(parent, url, options);
101          //props.setIncludedURL(url);
102          
103          Object result=urlPropsLoader.loadProperties(parent, url, options);
104          
105          if (result instanceof EProperties) {
106             ((EProperties)result).setIncludedURL(url);
107          }
108          return result;
109       }
110       
111       /** 
112        * Here we are supporting an include options syntax.  All possible
113        * options should have sensible default values in code.  It is up to 
114        * each include processor to respect or ignore these options.
115        * 
116        * The 2 key options as this point are (defaults in bold):
117        *    parse=[TRUE|false]
118        *    failonerrror=[TRUE|false]
119        *  
120        * The syntax is simple: 
121        * 
122        * prebill.sql=[sql/prebill.sql|parse=false,failonerror=true]
123        * 
124        * Pipe is the delimiter, followed by a csv list of key=value options.
125        * 
126        * This could also be used to facilitate JDBC based includes - 
127        * with the URL being something like:  
128        *   [jdbc://.......|user=foo,pass=bar,keycol=key,valcol=value]
129        */
130       EProperties parseOptionsFromIncludeURL(String url) {
131          // this can and very often will be an empty set of options.
132          
133          EProperties options=new EProperties();
134          
135          int pipeindex=url.indexOf('|');
136          if (pipeindex > 0) {
137             options.put("original.url", url);
138             String opString=url.substring(pipeindex+1);
139             
140             StringTokenizer st=new StringTokenizer(opString, ",", false);
141             
142             while(st.hasMoreElements()) {
143                String kvp=st.nextToken();
144                
145                int equalsindex=kvp.indexOf("=");
146                if (equalsindex == -1) 
147                   options.putBoolean(kvp, true);
148                else {
149                   String key=kvp.substring(0, equalsindex);
150                   String value=kvp.substring(equalsindex+1);
151                   options.put(key, value);
152                }
153             }
154             log.info("Include options: \n"+options.list(2));
155          }
156          
157          return options;
158       }
159    }
