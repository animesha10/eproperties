010    /**
011     *
012     */
013    public class MethodPropertiesLoader implements PropertiesLoader {
014       static Log log=LogFactory.getLog(MethodPropertiesLoader.class);
015       
016       // FIXME: Add method execution URL handler.  Like this:
017       // [method://java.net.InetAddress.getLocalHost().getHostName()]
018       // this would return a String rather than a nested properties
019       // object.  Which could be difficult.  
020       // Could make some implicit assumptions about the method:// processor
021       //   - Strings would be single value
022       //   - List or String[] would be converted to List<String> using toString?
023       //   - Properties or Map or Hashtable would be convered to nested EProperties()
024       //   - Object could be converted to an EProperties object by introspecting
025       //     for public accessors and/or methods
026       //     This could be interesting.  So if a method returned URL, 
027       //     we could either use url.toString(), or 
028       //     introspect into URL and getFile(), getHost(), getPort(), and getProtocol() would
029       //     result in:
030       //     foo={
031       //        file=(result from getFile())
032       //        host=(result from getHost())
033       //        # etc, you get the picture.
034       //     }
035       
036       /** */
037       public boolean acceptsURL(String s) {
038          if (s.toLowerCase().startsWith("method://")) {
039             return true;
040          }
041          return false;
042       }
043    
044       /** */
045       public Object loadProperties(EProperties parent, String url, 
046             EProperties options) {
047          EProperties props=new EProperties(parent);
048          //props.addAll(System.getProperties());
049          
050          // parse the URL: 
051          // strip off method://
052          
053          url=url.substring("method://".length());
054          
055          // java.lang.System.getenv()
056          
057          // parse the classs name and method name.
058          String className=url.substring(0, url.lastIndexOf("."));
059          String methodName=url.substring(url.lastIndexOf(".")+1);
060          
061          if (methodName.endsWith("()"))
062             methodName=methodName.substring(0, methodName.length()-2);
063          
064          try {
065             Class clazz=Class.forName(className);
066             
067             Method method=clazz.getMethod(methodName);
068             
069             Object obj=method.invoke(null);
070             
071             if (obj == null) {
072                log.error("Method include of "+className+"."+methodName+" returned null.");
073             }
074             
075             if (obj instanceof Map) {
076                Map map=(Map)obj;
077                
078                Set keys=map.keySet();
079                int count=0;
080                for(Object key:keys) {
081                   count++;
082                   Object value=map.get(key);
083                   
084                   props.put(key.toString(), value.toString());
085                }
086                
087                log.debug("Loaded "+count+" properties with MethodLoader.");
088             } else {
089                log.error("Don't know how to process method include of object type "+obj.getClass().getName());
090             }
091          } catch (Exception ex) {
092             throw new RuntimeException("Error processing method:// inclusion.", ex);
093          }
094          
095          //if (true)
096          //   throw new RuntimeException("Error method:// inclusion not implemented.");
097          
098          return props;
099       }
100    }
