001    package net.jmatrix.eproperties.cli;
002    
003    import java.io.File;
004    import java.net.URL;
005    import java.util.logging.*;
006    
007    import net.jmatrix.eproperties.*;
008    import net.jmatrix.eproperties.parser.EPropertiesParser;
009    import net.jmatrix.eproperties.utils.JDK14LogConfig;
010    
011    import org.apache.commons.logging.*;
012    
013    /**
014     * A simple command line interface to the EProperties system. 
015     *
016     */
017    public class EPropertiesCLI {
018       static Log log=LogFactory.getLog(EPropertiesCLI.class);
019       static String usage=
020          "EPropertiesCLI [-debug] <Props.URL> \n"+
021          "  Props.URL: this is required. It can be a relatve file path or a full http/file URL.";
022       
023       /** */
024       public static void main(String[] args) throws Exception {
025          JDK14LogConfig.startup();
026          ArgParser ap=new ArgParser(args);
027          
028          String surl=ap.getLastArg();
029          
030          if (surl == null) {
031             System.out.println (usage);
032             System.exit(1);
033          }
034          
035          if (ap.getBooleanArg("-debug")) {
036             EProperties.debug=true;
037             System.out.println ("Setting root logger to debug.");
038             Logger logger=Logger.getLogger("");
039             logger.setLevel(Level.ALL);
040             
041             logger=Logger.getLogger(EProperties.class.getName());
042             logger.setLevel(Level.FINE);
043             
044             logger=Logger.getLogger(SubstitutionProcessor.class.getName());
045             logger.setLevel(Level.INFO);
046             
047             logger=Logger.getLogger(EPropertiesParser.class.getName());
048             logger.setLevel(Level.ALL);
049          } else {
050             Logger logger=Logger.getLogger("");
051             logger.setLevel(Level.INFO);
052          }
053          
054          URL url=null;
055          
056          if (surl.indexOf("://") != -1) {
057             System.out.println ("Constructing URL as URL");
058             url=new URL(surl);
059          } else {
060             System.out.println ("Constructing URL as File");
061             url=(new File(surl)).toURI().toURL();
062          }
063          System.out.println ("Loading with "+url);
064          
065          EProperties props=new EProperties();
066          
067          System.out.println ("\n********************  BEGIN LOAD  *******************");
068          props.load(url);
069          System.out.println ("********************   END LOAD  *******************\n");
070          
071          
072          
073          
074          System.out.println ("size: "+props.size());
075          System.out.println ("keys: "+props.getKeys().size());
076          
077    //      System.out.println ("override_sub="+props.getString("OVERRIDE_SUB"));
078    //      System.out.println();
079    //      
080    //      System.out.println ("find override_sub="+props.findProperty("OVERRIDE_SUB"));
081    //      System.out.println();
082    //      
083    //      System.out.println ("override="+props.getString("OVERRIDE"));
084          
085          System.out.println ("---- listing ----");
086          if (ap.getBooleanArg("-flatten", false))
087             props.flatten(".").list(System.out);
088          else
089             props.list(System.out);
090          System.out.println ("---- end listing ----");
091          
092          props.superList();
093       }
094    }
