031       /** */
032       public static void diff(String propsFile, ArgParser ap) throws Exception {
033          
034          String format=ap.getStringArg("-format", "20|30");
035          if (format.indexOf("|") == -1) 
036             throw new Exception("Invalid format '"+format+"'.  Format must be int|int.");
037          
038          String keysizeString=format.substring(0, format.indexOf("|"));
039          String valuesizeString=format.substring(format.indexOf("|")+1);
040          
041          int keysize=Integer.parseInt(keysizeString);
042          int valuesize=Integer.parseInt(valuesizeString);
043          
044          
045          try {
046             EProperties eProps=new EProperties(); // read(propsFile);
047             eProps.load(propsFile);
048             
049             Properties juProps=new Properties(); // read(con);
050             juProps.load(new FileReader(propsFile));
051             
052             List<String> allKeys=new ArrayList<String>();
053             
054             Iterator keys=eProps.keySet().iterator();
055             while (keys.hasNext()) {
056                allKeys.add(keys.next().toString());
057             }
058             
059             keys=juProps.keySet().iterator();
060             while (keys.hasNext()) {
061                String key=keys.next().toString();
062                if (!allKeys.contains(key))
063                   allKeys.add(key);
064             }
065             
066             Collections.sort(allKeys);
067             
068             StringBuilder header=new StringBuilder();
069             header.append(truncPad("   key   ", keysize, ' '));
070             header.append("|");
071             header.append(truncPad("       EProperties value", valuesize, ' '));
072             header.append("|");
073             header.append(truncPad("  java.util.Properties value", valuesize, ' '));
074             System.out.println(header);
075             
076             for (int i=0; i<allKeys.size(); i++) {
077                String key=allKeys.get(i);
078                String epVal=eProps.getProperty(key);
079                String juVal=juProps.getProperty(key);
080                
081                if (epVal == null)
082                   epVal="null";
083                if (juVal == null)
084                   juVal="null";
085                
086                if (!epVal.equals(juVal))
087                   key="*"+key;
088                
089                StringBuilder line=new StringBuilder();
090                line.append(truncPad(key, keysize, '-'));
091                
092                line.append("|");
093                line.append(truncPad(epVal, valuesize, ' '));
094                line.append("|");
095                line.append(truncPad(juVal, valuesize, ' '));
096                
097                System.out.println (line);
098             }
099             
100          } finally {
101             
102          }
103       }

111       /** returns a string of the exact length, by either truncating or
112        * padding.
113        */
114       static String truncPad(String s, int len, char padChar) {
115          if (s.length() == len || len == -1)
116             return s;
117          else if (s.length() < len) {
118             StringBuilder pad=new StringBuilder();
119             for (int i=0; i<len-s.length(); i++) {
120                pad.append(padChar);
121             }
122             return s + pad.toString();
123          } else {
124             return s.substring(0, len);
125          }
126       }
