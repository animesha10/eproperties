012    /**
013     * Loads properties from file, http(s), or classpath locations. 
014     */
015    public class URLPropertiesLoader implements PropertiesLoader {
016       static Log log=LogFactory.getLog(URLPropertiesLoader.class);
017       static CacheManager cacheManager=CacheManager.getInstance();
018       
019       /** */
020       public boolean acceptsURL(String s) {
021          if (s == null)
022             return false;
023          
024          if (s.toLowerCase().startsWith("file:/") ||
025              s.toLowerCase().startsWith("http://") ||
026              s.toLowerCase().startsWith("https://") ||
027              s.toLowerCase().startsWith("classpath:/")) {
028             return true;
029          }
030          return false;
031       }
032    
033       /** */
034       public Object loadProperties(EProperties parent, String surl, 
035             EProperties options) {
036    
037          Object result=null;
038    
039          
040          //surl=surl.toLowerCase();
041          
042          // Translate classpath:// URLs to Java internal URLs.
043          surl=URLUtil.convertClasspathURL(surl);
044          String lcurl=surl.toLowerCase();
045    //      if (surl.startsWith("classpath:/")) {
046    //         System.out.println ("Found classpath URL: "+surl);
047    //         // First strip off the stuff that I created - 'classpath://'
048    //         String resourcepath=surl.substring("classpath:/".length());
049    //         System.out.println ("Resource path: "+resourcepath);
050    //         URL url=URLPropertiesLoader.class.getResource(resourcepath);
051    //         
052    //         if (url == null) {
053    //            throw new RuntimeException("Cannot find resource in classpath for URL "+surl);
054    //         }
055    //         System.out.println (surl+" -> "+url.toExternalForm());
056    //         surl=url.toExternalForm(); // this will start with either file:// or jar:file:/
057    //      }
058    //      
059          if (lcurl.startsWith("http://") || lcurl.startsWith("https://") ||
060              lcurl.startsWith("file:/") || lcurl.startsWith("jar:file:/")) {
061             log.debug("Loading as absolute URL: "+surl); 
062             result=loadFromURL(surl, options, parent);
063          } else if (surl.startsWith("/")) { 
064             log.debug("Loading as absolute file: "+surl);
065             // ok, just try to open it as a local file.
066             File f=new File(surl);
067             if (f.exists()) {
068                try {
069                   result=loadFromURL(f.toURI().toURL().toExternalForm(), options, parent);
070                   //props.load(f.toURI().toURL());
071                } catch (IOException ex) {
072                   if (options.getBoolean("failonerror", true)) {
073                      throw new 
074                      RuntimeException("Error loading included properties from '"+
075                            surl+"'", ex);
076                   } else {
077                      log.warn("Cannot find file '"+surl+"', failonerror=false");
078                   }
079                }
080             } else {
081                // the file does not exist.  
082                if (options.getBoolean("failonerror", true)) {
083                   throw new RuntimeException("Cannot find file '"+surl+"' to include.");
084                } else {
085                   log.warn("Cannot find file '"+surl+"' to include. failonerror=false");
086                }
087             }
088          } else {
089             // ok, the URL does not start with [http:// | https:// | file://], so
090             // let's assume it is a relative reference to the most recent parent.
091             // This means that if the parent is:
092             //   file://../config/config.properties, and the include url is 'include.properties'
093             // then the include should be processed from that relative location:
094             //   file://../config/include.properties
095             //
096             // Similarly, if the parent URL is:
097             //    http://jmatrix.net/properties/public/foo.properties 
098             // and the url is:
099             //    '../bar/baz.properties'
100             // then the include should be
101             //    http://jmatrix.net/properties/public/../bar/baz.properties
102             
103             URL parentURL=parent.findSourceURL();
104             
105             log.debug("Loading as relative URL, parent URL is '"+parentURL+"'");
106             
107             if (parentURL == null) {
108                // ok, just try to open it as a local file.
109                File f=new File(surl);
110                if (f.exists()) {
111                   try {
112                      result=loadFromURL(f.toURI().toURL().toExternalForm(), options, parent);
113                      //props.load(f.toURI().toURL());
114                   } catch (IOException ex) {
115                      if (options.getBoolean("failonerror", true)) {
116                         throw new 
117                         RuntimeException("Error loading included properties from '"+
118                               surl+"'", ex);
119                      } else {
120                         log.warn("Cannot find file '"+surl+"', failonerror=false");
121                      }
122                   }
123                } else {
124                   // the file does not exist.  
125                   if (options.getBoolean("failonerror", true)) {
126                      throw new RuntimeException("Cannot find file '"+surl+"' to include.");
127                   } else {
128                      log.warn("Cannot find file '"+surl+"' to include. failonerror=false");
129                   }
130                }
131             } else {
132                // create relative URL string
133                String workingSURL=parentURL.toString();
134                workingSURL=workingSURL.substring(0, workingSURL.lastIndexOf("/")+1)+surl;
135                log.debug("Loading w/ rel URL: '"+workingSURL+"'");
136                result=loadFromURL(workingSURL, options, parent);
137    //            try {
138    //               URL url=new URL(workingSURL);
139    //               
140    //               props.load(url);
141    //            } catch (IOException ex) {
142    //               throw new 
143    //               RuntimeException("Error loading included properties from '"+surl+"'", ex);
144    //            }
145             }
146          }
147          
148          return result;
149       }

152       /** 
153        * Returns either a String or an EProperties object, dependent on 
154        * whether the parse property is true or false.
155        * 
156        * */
157       private Object loadFromURL(String surl, EProperties options, EProperties parent) {
158          Object result=null;
159          try {
160             URL url=new URL(surl);
161             if (options.getBoolean("parse", true)) {
162                EProperties props=new EProperties(parent);
163                props.load(url);
164                result=props;
165             } else {
166                // load it as a string
167                //InputStream is=url.openStream();
168                InputStream is=cacheManager.getInputStream(url);
169                try {
170                   String value=StreamUtil.readToString(is);
171                   result=value;
172                } finally {
173                   if (is != null)
174                      is.close();
175                }
176             }
177          } catch (IOException ex) {
178             
179             if (options.getBoolean("failonerror", true)) {
180                throw new 
181                RuntimeException("Error loading included properties from '"+surl+"'", ex);
182             } else {
183                log.info("Cannot load properties from  '"+surl+"', failonerror=false");
184             }
185          }
186          return result;
187       }
