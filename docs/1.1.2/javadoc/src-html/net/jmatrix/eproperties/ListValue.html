001    package net.jmatrix.eproperties;
002    
003    import java.util.*;
004    
005    /** 
006     * Represents a dynamic list value.
007     *
008     */
009    public class ListValue implements Value<List<String>> {
010       List<String> persistentValue=null;
011       EProperties owner=null;
012       
013       public ListValue(List<String> l, EProperties p) {
014          persistentValue=l;
015          owner=p;
016       }
017       
018       public String toString() {
019          return getRuntimeValue().toString();
020       }
021       
022       /** Returns a List<String> that has not been passed through the 
023        * SubstitutionProcessor. */
024       public List<String> getPersistentValue() {
025          return persistentValue;
026       }
027    
028       public List<String> getRuntimeValue() {
029          // FIXME: in the future, the choice of static vs. dynamic substitution
030          //        should be statically configurable.
031          
032          // perform dynamic substitution.
033          int size=persistentValue.size();
034          List<String> newList=new ArrayList<String>(size);
035          
036          for (int i=0; i<size; i++) {
037             newList.add(SubstitutionProcessor.processSubstitution(persistentValue.get(i), owner));
038          }
039          
040          return newList;
041       }
042    
043       @Override
044       public void setOwner(EProperties p) {
045          owner=p;
046       }
047    }
