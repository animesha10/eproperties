001    package net.jmatrix.util;
002    
003    
004    import java.util.*;
005    import java.lang.reflect.*;
006    
007    import net.jmatrix.eproperties.EProperties;
008    
009    import java.util.logging.*;
010    import static java.util.logging.Level.*;
011    
012    
013    /**
014     * The InstanceFactory builds instances of classes based on 
015     * properties files.  These can then be cast to appropriate types
016     * as necessary.
017     */
018    public final class InstanceFactory 
019    {
020       static final Logger log=Logger.getLogger(InstanceFactory.class.getName());
021       
022       /** */
023       static Map<String, Constructor> constructorCache=new HashMap();
024    
025       /** 
026        * Builds a new instance of a class based on an EProperties file.
027        * The properties MUST CONTAIN: 
028        *  - 'class' which is the fully qualified (with package) class name 
029        *    of the implementing class.
030        * 
031        * The class defined above MUST be available in the current JVM
032        * CLASSPATH.  Ie, Class.forName(class) must resolve.
033        * 
034        * The Class defined above MUST have at least 1 constructor that 
035        * take an EProperties object as a parameter.  These properties
036        * are passed to the implementing class's constructor.
037        *
038        * Users of this class will always cast the resulting object 
039        * to the appropriate type.
040        *
041        * @param EProperties The properties object which defines the class
042        * and any parameters for its construction.
043        * @return An Object, which is an instance of the class requested.
044        */
045       public static Object buildInstance(EProperties props) 
046          throws InstantiationException, ClassNotFoundException, 
047          NoSuchMethodException, IllegalAccessException, 
048          InvocationTargetException {
049          // This method makes 3 assumptions:
050          // 1) The properies file contains a string property 
051          //    called 'class' - which is the class name of the 
052          //    implmementation class
053          // 2) The class specified in 'class' exists in the current 
054          //    classpath.
055          // 3) The class has a public constructor that takes an 
056          //    EProperties object.
057    
058          Object target=null;
059          String className=props.getString("class");
060    
061          log.log(FINE, "InstanceFactory: building instance of "+className);
062          if (className == null) {
063             throw new RuntimeException("InstanceFactory: ERROR.  Property "+
064                                        "'class' is required.  Inbound "+
065                                        "properties does not contain 'class'");
066          }
067    
068          if (className == null) {
069             throw new 
070             IllegalArgumentException("InstanceFactoryr: Required property "+
071                                      "class which defines the "+
072                                      "implementation class is null.  Props must "+
073                                      "contain 'class' property.");
074          }
075          Constructor constructor=getConstructor(className);
076    
077          target=constructor.newInstance(props);
078    
079          return target;
080       }
081       
082       /** */
083       public static Object runtimeBuildInstance(EProperties p) {
084          try {
085             return buildInstance(p);
086          } catch (Exception ex) {
087             throw new RuntimeException("Error building instance!", ex);
088          }
089       }
090    
091       /** */
092       private static final Constructor getConstructor(String className) 
093          throws InstantiationException, ClassNotFoundException, 
094          NoSuchMethodException {
095          Constructor constructor=constructorCache.get(className);
096          
097          if (constructor == null) {
098             log.log(FINE, "InstanceFactory: caching constructor for '"+className+"'");
099             constructor=buildConstructor(className);
100          }
101          return constructor;
102       }
103    
104       /** */
105       private static final Constructor buildConstructor(String className) 
106          throws InstantiationException, ClassNotFoundException, 
107          NoSuchMethodException {
108          Class clazz=Class.forName(className);
109             
110          // I apologize in advance for this obscure syntax.  Bemo.
111          Constructor constructor=clazz.getConstructor(new Class[]{EProperties.class});
112          constructorCache.put(className, constructor);
113          return constructor;
114       }
115    }
