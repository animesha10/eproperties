001    package net.jmatrix.eproperties.cli;
002    
003    import java.io.File;
004    import java.util.List;
005    
006    import net.jmatrix.eproperties.*;
007    
008    public class EPropertify {
009    
010       /**
011        * @param args
012        */
013       public static void main(String[] args) throws Exception  {
014          ArgParser ap=new ArgParser(args);
015          
016          // should be
017          // epropertify -i <infile/url> -o <outfile> -d <delimeter>
018          
019          String in=ap.getStringArg("-i");
020          String out=ap.getStringArg("-o");
021          String delim=ap.getStringArg("-d", ".");
022          
023          EProperties inProps=new EProperties();
024          inProps.load(in);
025          inProps=inProps.flatten(delim);
026          
027          EProperties outProps=new EProperties();
028          File outFile=new File(out);
029          
030          
031          List<Key> keys=inProps.getKeys();
032          for (Key key:keys) {
033             String skey=key.toString();
034             String newkey=skey.replace(delim, "->");
035             System.out.println (skey+" ->   "+newkey);
036             
037             outProps.put(newkey, inProps.get(skey));
038          }
039          
040          outProps.save(outFile);
041       }
042    }
