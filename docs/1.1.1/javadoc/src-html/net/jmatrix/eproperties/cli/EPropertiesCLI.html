001    package net.jmatrix.eproperties.cli;
002    
003    import java.io.File;
004    import java.net.URL;
005    import java.util.logging.*;
006    
007    import net.jmatrix.eproperties.EProperties;
008    import net.jmatrix.eproperties.utils.JDK14LogConfig;
009    
010    import org.apache.commons.logging.*;
011    
012    /**
013     * A simple command line interface to the EProperties system. 
014     *
015     */
016    public class EPropertiesCLI {
017       static Log log=LogFactory.getLog(EPropertiesCLI.class);
018       static String usage=
019          "EPropertiesCLI [-debug] <Props.URL> \n"+
020          "  Props.URL: this is required. It can be a relatve file path or a full http/file URL.";
021       
022       /** */
023       public static void main(String[] args) throws Exception {
024          JDK14LogConfig.startup();
025          ArgParser ap=new ArgParser(args);
026          
027          String surl=ap.getLastArg();
028          
029          if (surl == null) {
030             System.out.println (usage);
031             System.exit(1);
032          }
033          
034          if (ap.getBooleanArg("-debug")) {
035             System.out.println ("Setting root logger to debug.");
036             Logger logger=Logger.getLogger("");
037             logger.setLevel(Level.FINEST);
038             
039             logger=Logger.getLogger(EProperties.class.getName());
040             logger.setLevel(Level.FINE);
041          } else {
042             Logger logger=Logger.getLogger("");
043             logger.setLevel(Level.INFO);
044          }
045          
046          URL url=null;
047          
048          if (surl.indexOf("://") != -1) {
049             System.out.println ("Constructing URL as URL");
050             url=new URL(surl);
051          } else {
052             System.out.println ("Constructing URL as File");
053             url=(new File(surl)).toURI().toURL();
054          }
055          System.out.println ("Loading with "+url);
056          
057          EProperties props=new EProperties();
058          
059          props.load(url);
060          
061          
062          System.out.println ("---- listing ----");
063          if (ap.getBooleanArg("-flatten", false))
064             props.flatten(".").list(System.out);
065          else
066             props.list(System.out);
067          System.out.println ("---- end listing ----");
068       }
069    }
